# Read and select variables
df = pd.read_csv("supermarkets_data_enriched.csv") 
selected_columns = ['id', 'brand', 'lat', 'lon'] 
df = df[selected_columns] 

# Subset
df_sub = df.loc[df['brand'].isin(['Coop', 'Migros'])]  # Corrected syntax for subsetting DataFrame

# Colors
colors = {'Coop': 'green', 'Migros': 'orange'}  # Corrected syntax for dictionary

# Plot
plt.scatter(df_sub['lon'], df_sub['lat'], c=df_sub['brand'].map(colors))
plt.title('Supermarket Locations by Brand')
plt.xlabel('Longitude')
plt.ylabel('Latitude')
plt.show()
